9/*
10 * The permission system aims to allow a general-purpose implementation for
11 * all common hardware register usages. This includes:
12 * - Trivial Cases: Read, Write, Read & Write
13 *
14 * - Read & Write accesses that route to different values/functions of a chip.
15 * (e.g. serial communication, where read/write corresponds to RX/TX)
16 * - Read to clear, write to clear. This does not directly affect the access,
17 * but can be used to implement a software shadow register for flags
18 * (ORing the read value into a shadow register instead of overwriting).
19 * - Registers with default values that are not known (e.g. Factory configuration
20 * values that should not be overwritten by default).
21 */
22
23#ifndef TMC_HELPERS_REGISTERACCESS_H
24#define TMC_HELPERS_REGISTERACCESS_H

26// Register access bits
27/* Lower nibble is used for read/write, higher nibble is used for
28 * special case registers. This makes it easy to identify the read/write
29 * part of the permissions in a hexadecimal permission number.
30 * The dirty bit will only ever be set at runtime, so we keep the easily
31 * readable lower nibble.
32 */
33#define TMC_ACCESS_NONE 0x00
34
35#define TMC_ACCESS_READ 0x01
36#define TMC_ACCESS_WRITE 0x02
37 // 0x04 is currently unused
38#define TMC_ACCESS_DIRTY 0x08 // Register has been written since reset -> shadow register is valid for restore
39
40// Special Register bits
41#define TMC_ACCESS_RW_SPECIAL 0x10 // Read and write are independent - different values and/or different functions
42#define TMC_ACCESS_FLAGS 0x20 // Register has read or write to clear flags.
43#define TMC_ACCESS_HW_PRESET 0x40 // Register has hardware presets (e.g. Factory calibrations) - do not write a default value
44 // 0x80 is currently unused
45
46// Permission combinations
47#define TMC_ACCESS_RW (TMC_ACCESS_READ | TMC_ACCESS_WRITE) // 0x03 - Read and write
48#define TMC_ACCESS_RW_SEPARATE (TMC_ACCESS_RW | TMC_ACCESS_RW_SPECIAL) // 0x13 - Read and write, with separate values/functions
49#define TMC_ACCESS_R_FLAGS (TMC_ACCESS_READ | TMC_ACCESS_FLAGS) // 0x21 - Read, has flags (read to clear)
50#define TMC_ACCESS_RW_FLAGS (TMC_ACCESS_RW | TMC_ACCESS_FLAGS) // 0x23 - Read and write, has flags (read or write to clear)
51#define TMC_ACCESS_W_PRESET (TMC_ACCESS_WRITE | TMC_ACCESS_HW_PRESET) // 0x42 - Write, has hardware preset - skipped in reset routine
52#define TMC_ACCESS_RW_PRESET (TMC_ACCESS_RW | TMC_ACCESS_HW_PRESET) // 0x43 - Read and write, has hardware presets - skipped in reset routine
53
54// Helper macros
55#define TMC_IS_READABLE(x) ((x) & TMC_ACCESS_READ)
56#define TMC_IS_WRITABLE(x) ((x) & TMC_ACCESS_WRITE)
57#define TMC_IS_DIRTY(x) ((x) & TMC_ACCESS_DIRTY)
58#define TMC_IS_PRESET(x) ((x) & TMC_ACCESS_HW_PRESET)
59#define TMC_IS_RESETTABLE(x) (((x) & (TMC_ACCESS_W_PRESET)) == TMC_ACCESS_WRITE) // Write bit set, Hardware preset bit not set
60#define TMC_IS_RESTORABLE(x) (((x) & TMC_ACCESS_WRITE) && (!(x & TMC_ACCESS_HW_PRESET) || (x & TMC_ACCESS_DIRTY))) // Write bit set, if it's a hardware preset register, it needs to be dirty

62// Struct for listing registers that have constant contents which we cannot
63// obtain by reading them due to the register not being read-back.

70// Helper define:
71// Most register permission arrays are initialized with 128 values.
72// In those fields its quite hard to have an easy overview of available
73// registers. For that, ____ is defined to 0, since 4 underscores are
74// very easy to distinguish from the 2-digit hexadecimal values.
75// This way, the used registers (permission != ACCESS_NONE) are easily spotted
76// amongst unused (permission == ACCESS_NONE) registers.
77#define ____ 0x00
78
79// Helper define:
80// Default reset values are not used if the corresponding register has a
81// hardware preset. Since this is not directly visible in the default
82// register reset values array, N_A is used as an indicator for a preset
83// value, where any value will be ignored anyways (N_A: not available).
84#define N_A 0
