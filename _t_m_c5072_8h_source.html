68 int home(float maxHomingDistance = 15, float switchOffset = 0.1, float finalSpeed = 0.1, float switchBackDistance = 0.25, float switchExtraDistance = 0.125)
69 {
70 enum State {
71 start,
72 startMovingTowardsSwitch,
73 movingTowardsSwitch,
74 startMovingAwayFromSwitch,
75 movingAwayFromSwitch,
76 startFinalMoveTowardsSwitch,
77 finalMoveTowardsSwitch,
78 movingToFinalZero,
79 homeFinished,
80 };
81 State state = start;
82 boolean sw;
83 uint32_t xlatch;
84 commWrite(TMC5072_SWMODE(whichMotor) + TMC_WRITE_BIT, ((true << TMC5072_LATCH_L_ACTIVE_SHIFT) | (true << TMC5072_EN_SOFTSTOP_SHIFT) | (true << TMC5072_STOP_L_ENABLE_SHIFT))); // latch xactual into xlatch when left switch activated
85
86 while (true) { // returned from to exit
87 switch (state) {
88 case start:
89 setXActualRaw(0);
90 moveToPositionRaw(0);
91 if (getStopSwitchStatus(true)) { // already at home
92 xlatch = commRead(TMC5072_XLATCH(whichMotor));
93 state = startMovingAwayFromSwitch;
94 } else {
95 state = startMovingTowardsSwitch;
96 }
97 break;
98 case startMovingTowardsSwitch: // moving towards switch
99 moveToPosition(-abs(maxHomingDistance)); // enough to hit switch, but won't go forever
100 state = movingTowardsSwitch;
101 break;
102 case movingTowardsSwitch:
103 do {
104 delay(1);
105 sw = getStopSwitchStatus(true);
106 } while (!sw && isNotAtTarget(true));
107 if (sw) {
108 xlatch = commRead(TMC5072_XLATCH(whichMotor));
109 state = startMovingAwayFromSwitch;
110 } else {
111 return 1; // didn't hit switch
112 }
113 break;
114 case startMovingAwayFromSwitch:
115 moveToPositionRaw(xlatch + abs(switchBackDistance) / positionScaler);
116 state = movingAwayFromSwitch;
117 break;
118 case movingAwayFromSwitch:
119 do {
120 delay(1);
121 } while (isNotAtTarget(true));
122 sw = getStopSwitchStatus(true);
123 if (sw) { // switch still active for some reason, error
124 return 2; // switch didn't deactivate
125 } else {
126 state = startFinalMoveTowardsSwitch;
127 }
128 break;
129 case startFinalMoveTowardsSwitch:
130
131 setVels(finalSpeed);
132 moveToPositionRaw(xlatch - abs(switchExtraDistance) / positionScaler);
133 state = finalMoveTowardsSwitch;
134 break;
135 case finalMoveTowardsSwitch:
136 do {
137 delay(1);
138 sw = getStopSwitchStatus(true);
139 } while (!sw && isNotAtTarget(true));
140 if (sw) {
141 xlatch = commRead(TMC5072_XACTUAL(whichMotor));
142 setVels(1); // reset velocity
143 moveToPositionRaw(xlatch + switchOffset / positionScaler);
144 state = movingToFinalZero;
145 } else {
146 return 3; // didn't hit switch
147 }
148 break;
149 case movingToFinalZero:
150 commWrite(TMC5072_SWMODE(whichMotor) + TMC_WRITE_BIT, 0); // latch xactual into xlatch when left switch activated
151 do {
152 delay(1);
153 } while (isNotAtTarget(true));
154 setXActualRaw(0);
155 moveToPositionRaw(0);
156 do {
157 delay(1);
158 } while (isNotAtTarget(true));
159 if (getStopSwitchStatus(true)) {
160 return 4; // didn't move off the switch
161 }
162 commWrite(TMC5072_SWMODE(whichMotor) + TMC_WRITE_BIT, ((true << TMC5072_LATCH_L_ACTIVE_SHIFT) | (false << TMC5072_EN_SOFTSTOP_SHIFT) | (true << TMC5072_STOP_L_ENABLE_SHIFT))); // latch xactual into xlatch when left switch activated
163 return 0;
164 break;
165 }
166 }
