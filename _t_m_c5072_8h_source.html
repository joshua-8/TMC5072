84 int home(float maxHomingDistance = 15, float switchOffset = 1, float finalSpeed = 0.1, float switchBackDistance = 0.25, float switchExtraDistance = 0.125, unsigned long smallMovementTimeoutMS = 5000, unsigned long firstMovementTimeoutMS = 15000)

86 enum State {
87 start,
88 startMovingTowardsSwitch,
89 movingTowardsSwitch,
90 startMovingAwayFromSwitch,
91 movingAwayFromSwitch,
92 startFinalMoveTowardsSwitch,
93 finalMoveTowardsSwitch,
94 movingToFinalZero,
95 homeFinished,
96 };
97 State state = start;
98 boolean sw;
99 uint32_t xlatch;
100 commWrite(TMC5072_SWMODE(whichMotor) + TMC_WRITE_BIT, ((true << TMC5072_LATCH_L_ACTIVE_SHIFT) | (true << TMC5072_EN_SOFTSTOP_SHIFT) | (true << TMC5072_STOP_L_ENABLE_SHIFT))); // latch xactual into xlatch when left switch activated
101
102 unsigned long stateStarted = millis();
103 while (true) { // returned from to exit
104 stateStarted = millis();
105 switch (state) {
106 case start:
107 setXActualRaw(0);
108 moveToPositionRaw(0);
109 if (getStopSwitchStatus(true)) { // already at home
110 xlatch = commRead(TMC5072_XLATCH(whichMotor));
111 state = startMovingAwayFromSwitch;
112 } else {
113 state = startMovingTowardsSwitch;
114 }
115 break;
116 case startMovingTowardsSwitch: // moving towards switch
117 moveToPosition(-abs(maxHomingDistance)); // enough to hit switch, but won't go forever
118 state = movingTowardsSwitch;
119 break;
120 case movingTowardsSwitch:
121 do {
122 delay(1);
123 sw = getStopSwitchStatus(true);
124 } while (!sw && isNotAtTarget(true) && (millis() - stateStarted) < firstMovementTimeoutMS);
125 if (sw) {
126 xlatch = commRead(TMC5072_XLATCH(whichMotor));
127 state = startMovingAwayFromSwitch;
128 } else {
129 return 1; // didn't hit switch
130 }
131 break;
132 case startMovingAwayFromSwitch:
133 moveToPositionRaw(xlatch + abs(switchBackDistance) / positionScaler);
134 state = movingAwayFromSwitch;
135 break;
136 case movingAwayFromSwitch:
137 do {
138 delay(1);
139 } while (isNotAtTarget(true) && (millis() - stateStarted) < smallMovementTimeoutMS);
140 sw = getStopSwitchStatus(true);
141 if (sw) { // switch still active for some reason, error
142 return 2; // switch didn't deactivate
143 } else {
144 state = startFinalMoveTowardsSwitch;
145 }
146 break;
147 case startFinalMoveTowardsSwitch:
148
149 setVels(finalSpeed);
150 moveToPositionRaw(xlatch - abs(switchExtraDistance) / positionScaler);
151 state = finalMoveTowardsSwitch;
152 break;
153 case finalMoveTowardsSwitch:
154 do {
155 delay(1);
156 sw = getStopSwitchStatus(true);
157 } while (!sw && isNotAtTarget(true) && (millis() - stateStarted) < smallMovementTimeoutMS);
158 if (sw) {
159 xlatch = commRead(TMC5072_XACTUAL(whichMotor));
160 resetToDefaultVelocitiesAndAccelerations();
161 moveToPositionRaw(xlatch + ((uint32_t)(switchOffset / positionScaler)));
162 state = movingToFinalZero;
163 } else {
164 return 3; // didn't hit switch
165 }
166 break;
167 case movingToFinalZero:
168 commWrite(TMC5072_SWMODE(whichMotor) + TMC_WRITE_BIT, 0); // latch xactual into xlatch when left switch activated
169 do {
170 delay(1);
171 } while (isNotAtTarget(true) && (millis() - stateStarted) < smallMovementTimeoutMS);
172 setXActualRaw(0);
173 moveToPositionRaw(0);
174 do {
175 delay(1);
176 } while (isNotAtTarget(true) && (millis() - stateStarted) < smallMovementTimeoutMS);
177 if (getStopSwitchStatus(true)) {
178 return 4; // didn't move off the switch
179 }
180 commWrite(TMC5072_SWMODE(whichMotor) + TMC_WRITE_BIT, ((true << TMC5072_LATCH_L_ACTIVE_SHIFT) | (false << TMC5072_EN_SOFTSTOP_SHIFT) | (true << TMC5072_STOP_L_ENABLE_SHIFT))); // latch xactual into xlatch when left switch activated
181 return 0;
182 break;
183 }
